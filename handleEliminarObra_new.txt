  const handleEliminarObra = async (id) => {
    const obra = obras.find(o => o.id === id);
    const confirmMessage =
      `¿Está seguro de cancelar la obra "${obra?.nombre || id}"?\n\n` +
      `La obra cambiará su estado a CANCELADO:\n\n` +
      `✓ Se conservan todos los datos históricos\n` +
      `✓ La obra no aparecerá en listados activos\n` +
      `✓ Se puede consultar en cualquier momento\n` +
      `✓ Los datos financieros se mantienen intactos\n\n` +
      `Esta acción es reversible cambiando el estado de la obra.`;

    if (!window.confirm(confirmMessage)) return;

    try {
      showNotification('Cancelando obra...', 'info');

      // Cambiar estado a CANCELADO en lugar de eliminar
      await handleCambiarEstado(id, 'CANCELADO');

      showNotification(
        `Obra "${obra?.nombre || id}" cancelada exitosamente.\n\nLa obra cambió su estado a CANCELADO y conserva todos sus datos.`,
        'success'
      );

      setSelectedObraId(null);

    } catch (error) {
      console.error('Error cancelando obra:', error);
      showNotification(
        `Error al cancelar la obra: ${error.message}\n\nIntente nuevamente o contacte al administrador.`,
        'error'
      );
    }
  };
